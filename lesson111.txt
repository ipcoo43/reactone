[ if 조건문 ]
- 조건을 비교한 다음, 조건의 참 거짓에 따라 다른 명령 수행

if 조건문:
  참일 경우 수행할 문장들

a=3       # a 변수에 3이라는 객체가 연결되었다. namespace가 a 이다.
if a > 0: # a가 0보다 크면 True 이다.
  print('True')  # True라는 문장을 출력하라

a=[1,2,3,4,5]  # a 변수에 list 객체가 연결되었다
if 2 in a:     # 2가 list a에 속하면 True 이다.
  print('True')  # True라는 문장을 출력하라

[ for if ]
a=[1,2,3,4,5]  # a 변수에 list 객체가 연결되었다
for i in a:    # list a 중에서
  if i > 3:    # i가 3보다 큰 원소를
     print(i)  # i를 출력하라

[ if else ]
- 조건이 참일 경우 if 다음 문장 실행하고
- 거짓이면 else 다음 문장 실행

if 조건문 :
  참일 경우 실행할 문장들
else:
  거짓일 경우 시행할 문장들

a = 4     # a 변수에 4라는 객체를 연결하고
if a < 0: # a가 0보다 작으면
  print('0 보다 작다')  # 이 문장을 출력하고
else:     # a가 0보다 크거나 같으면
  print('0 보다 크거나 같다')  # 이 문장을 출력하라
# 그러면 0 보다 크거나 같다가 출력 될 것이다.

# 가지고 있는 돈은 4900원 이다.
# 5000원 이상이면 택시를 타고 가고
# 5000원 미만이면 지하철을 타고간다.
money = 4900  # money 변수에 4900 객체를 연결하고
if money >= 5000: # money가 5000보다 크거나 같으면
  print("택시 타고가자.") # 이 문장을 출력하고 
else:   # money가 5000보다 작으면
  print("지하철 타고가자.") # 이 문장을 출력하라
# 그러면 지하철을 타고가라는 문장이 출력 될 것 이다.

[ if elsif else ]
if 조건문 1:
  조건문1 만족 수행할 문장들
elif 조건문 2:
  조건문2 만족 수행할 문장들
else:
  그밖의 경우 수행할 문장들

a = ['오이', '상추', '고추']
b = ['사과', '수박']
c = ['닭', '소', '돼지']
d = '오이'
if d in a:       # 오이가 a 리스트 안에 있으면
  print("야채")  # 야채를 출력하라
elif d in b:     # 오이가 b 리스트 안에 있으면
  print("과일")  # 과일을 출력하라
elif d in c:     # 오이가 c 리스트 안에 있으면
  print("고기")  # 고기를 출력하라
else:            # 그 밖의 경우라면
  print("야채,과일,고기도 아니다") # 이 문장을 출력하라
# 그러면 야채가 출력 될 것이다.

d = '아이폰'
if d in a:       # 아이폰이 a 리스트 안에 있으면
  print("야채")  # 야채를 출력하라
elif d in b:     # 아이폰이 b 리스트 안에 있으면
  print("과일")  # 과일을 출력하라
elif d in c:     # 아이폰이 c 리스트 안에 있으면
  print("고기")  # 고기를 출력하라
else:            # 그 밖의 경우라면
  print("야채,과일,고기도 아니다") # 이 문장을 출력하라
# 그러면 "야채,과일,고기도 아니다"가 출력 될 것이다.

money = 0 # money 변수에 0 객체를 연결하고
if money >= 5000: # money가 5000보다 크거나 같으면
  print("택시 타고가자") # 이 문장을 실행하고
elif money > 1000: # money가 1000보다 크다면
  print("지하철 타고가자.") # 이 문장을 실행하고
else:  # 그 밖의 경우라면
  print("걸어가야겠다") # 이 문자을 실행하라
